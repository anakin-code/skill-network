private List<SkillExperienceResponse> createSkillExperiences(Profile profile) {
    Map<String, Long> skillDaysMap = new HashMap<>();

    for (Career career : profile.getCareers()) {
        LocalDate start = career.getStartTime();
        LocalDate end = career.getEndTime() == null
                ? LocalDate.now()
                : career.getEndTime();

        long days = ChronoUnit.DAYS.between(start, end);

        for (CareerSkill careerSkill : career.getCareerSkills()) {
            String skillName = careerSkill.getSkill().getSkillName();

            skillDaysMap.merge(skillName, days, Long::sum);
        }
    }

    return skillDaysMap.entrySet().stream()
            .map(entry -> {
                long totalDays = entry.getValue();
                int years = (int) (totalDays / 365);
                int months = (int) ((totalDays % 365) / 30);

                return new SkillExperienceResponse(
                        entry.getKey(),
                        years,
                        months
                );
            })
            .toList();
}

private ProfileResponse convertToProfileResponse(Profile profile) {
    List<CareerResponse> careers =
            convertToCareerResponses(profile.getCareers());

    List<SkillExperienceResponse> skillExperiences =
            createSkillExperiences(profile);

    return new ProfileResponse(
            profile.getHrid(),
            profile.getName(),
            profile.getMailAddress(),
            profile.getRank().getRankName(),
            profile.getFree(),
            careers,
            skillExperiences
    );
}
